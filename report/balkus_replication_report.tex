\section{Setup}
%======================================================================

\textbf{Problem.} Cross-fitting removes overfitting bias in DML \citep{chernozhukov2018}. With clustered data, \citet{chiang2021} say splits must respect clusters (K$^2$-fold). \citet{balkus2026} prove plain random splits (``as-IID'') work just as well, as long as cluster sizes stay bounded. But standard errors still need cluster adjustment.

\textbf{Gap.} The \texttt{DoubleML} library bundles splitting and SE estimation: \texttt{cluster\_cols} set $\to$ K$^2$-fold + CR SEs; unset $\to$ as-IID + IID SEs. The optimal combo (as-IID + CR SEs) is unavailable. We implement it ourselves.

\textbf{Three strategies:}
\begin{itemize}[leftmargin=1.5em, nosep]
  \item \textbf{No CF} --- no splitting, train = test (baseline)
  \item \textbf{As-IID} --- random $K$-fold, ignoring clusters
  \item \textbf{K$^2$-fold} --- cluster-aware splitting \citep{chiang2021}
\end{itemize}

\textbf{Three learners:}
\begin{itemize}[leftmargin=1.5em, nosep]
  \item \textbf{MARS} --- built-in GCV pruning, approximately Donsker (doesn't overfit)
  \item \textbf{Deep RF} --- fully grown trees, memorizes training data (not Donsker)
  \item \textbf{Lasso} --- L1 regularization keeps EP bias small (behaves Donsker)
\end{itemize}

\textbf{Metrics} \citep{morris2019}. Since we know the true $\theta_0$, we can check every aspect of estimation quality:
\begin{itemize}[leftmargin=1.5em, nosep]
  \item \textbf{Bias} $= \bar{\hat\theta} - \theta_0$ --- are estimates centered on the truth?
  \item \textbf{EmpSE} $= \text{SD}(\hat\theta)$ --- how much do estimates scatter across replications? (true uncertainty)
  \item \textbf{ModSE} $= \sqrt{\text{mean}(\widehat{SE}^2)}$ --- what the SE formula \emph{thinks} the uncertainty is
  \item \textbf{RelSE} $=$ ModSE/EmpSE --- do the reported SEs match reality? ($1.0$ = perfect, $<1$ = SEs too small)
  \item \textbf{Coverage} --- how often does the 95\% CI actually contain $\theta_0$?
\end{itemize}

\midrule

Key takeaways:
\begin{itemize}[leftmargin=1.5em, nosep]
  \item MARS and Lasso have near-zero bias without CF --- they're Donsker-like, cross-fitting is optional for them
  \item Deep RF without CF: systematic bias ($-0.040$), RelSE $= 0.40$ --- broken
  \item Cross-fitting fixes Deep RF (bias $\to +0.032$, RelSE $\to 0.97$), but EmpSE rises (0.059 $\to$ 0.095) because each fold sees less data
  \item K$^2$-fold causes data starvation: MARS EmpSE explodes from 0.054 to 1.163
\end{itemize}

\section{Boundary: When Theory Breaks}
%======================================================================

Balkus requires bounded cluster sizes. On a full $N \times M$ grid ($C = M \to \infty$), coverage drops 85\% $\to$ 71\% as the grid grows --- for \emph{both} strategies equally. The CR SE formula itself breaks, not the splitting. On bounded-cluster DGPs ($d=20$), both strategies hit 97\%.


%======================================================================
\section{Bottom Line}
%======================================================================

\begin{itemize}[leftmargin=1.5em, nosep]
  \item \textbf{Balkus holds.} As-IID $\geq$ K$^2$-fold everywhere. Cross-fitting only matters for reckless learners (Deep RF), not careful ones (MARS, Lasso).
  \item \textbf{The real issue is SEs.} CR SEs are essential with clustered data. When clusters grow unbounded, the CR formula itself breaks.
  \item \textbf{DoubleML should decouple splitting from SEs.} The optimal combo (as-IID + CR) isn't available.
  \item \textbf{Recipe:} As-IID cross-fitting + cluster-robust SEs. Always.
\end{itemize}
